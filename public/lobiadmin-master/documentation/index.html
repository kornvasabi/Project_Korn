Url routing

LobiAdmin support url routing for url hashes. Url routing is enabled by default but you can disable it by setting enableUrlRouting : false.

Url routing supports regexp. By default all pages are redirected into pages/hashName.html.

window.LobiAdminRoutes = {
    '(.+)' : 'pages/$1.html'
};

Redirect only dashboard

You can redirect only specific hashes by adding/removing routes from window.LobiAdminRoutes object.

window.LobiAdminRoutes = {
    'dash': 'pages/dashboard.html',
};
When url hash is dash page will be loaded from pages/dashboard.html

Redirect only specific hashes.

window.LobiAdminRoutes = {
    '(dashboard|calendar|profile)': 'pages/$1.html'
};

if hash is dashboard page will be loaded from pages/dashboard.html

if hash is calendar page will be loaded from pages/calendar.html

if hash is profile page will be loaded from pages/profile.html


If url router is not found the same hash is used for page loading. For example if LobiAdminRoutes looks like this

{
    '(dashboard|calendar|profile)': 'pages/$1.html'
};

and html link looks like this

<a href="#pages/timeline.html">Timeline</a>

page will be loaded from pages/timeline.html

You can have enableUrlRouting : true but used unrouted hashes also

window.LobiAdminRoutes = {
    '(dashboard|calendar|profile)': 'pages/$1.html'
};

    <li><a href="#dashboard">Dashboard</a></li>

    <li><a href="#profile">Profile</a></li>

LobiAdmin.js file

LobiAdmin.js file contains two classes: LobiAdmin and NavigationPanel.

When LobiAdmin webapp is loaded instances of these classes are created and saved in data attributes of $('body') and $(window.LobiAdminConfig.sidebarSelector) respectivelly. You can get instances of these classes by the following way: 

var lobiAdmin = $('body').data('lobiAdmin');
var menu = $(window.LobiAdminConfig.sidebarSelector).data('lobiAdmin.navMenu');
//Or
var menu = lobiAdmin.$navPanel;

//Initialize tooltips and popovers

//Prevent empty links click and the links which href="" attribute is current url hash value
$(document).on('click', 'a', function(ev){
    var $a = $(this);
    if ($a.attr('href') === '#' || window.location.hash === $a.attr('href')){
        ev.preventDefault();
    }
});
//Prevent showing disabled tabs
$(document).on('show.bs.tab', '.nav-tabs li.disabled>a, .nav-pills li.disabled>a', function(ev){
    ev.preventDefault();
});

//LobiBox default options
Lobibox.notify.OPTIONS = $.extend({}, Lobibox.notify.OPTIONS, {
    soundPath: 'sound/lobibox/'
});

/**
 * When this element is clicked email will be opened by the data-key="" attribute of clicked element
 */
$(document).on('click', LobiAdminConfig.openEmailViewSelector, function(){
    var $this = $(this);
    //If lobimail is already loaded
    if (window.location.hash === $this.attr('href')){
        var $mail = $('.lobimail').data('lobiMail');
        $mail.viewEmail($this.data('key'));
    }
    //If mailbox view is not loaded it will be loaded as soon as the link is clicked 
    //and we attach event listener. Before mailbox is finaly initialized we change default view option 
    else{
        $(document).on('beforeInit.lobiMail.1', '.lobimail', function(ev, $mail){
            $mail.setDefaultView('email', $this.data('key'));
            $(document).off('beforeInit.lobiMail.1', '.lobimail');
        });
    }
});
/**
 * When you click this element compose email interface will be opened
 */
$(document).on('click', LobiAdminConfig.composeEmailViewSelector, function(ev){
    var $this = $(this);
    //If lobimail is already loaded
    if (window.location.hash === $this.attr('href')) {
        var $mail = $('.lobimail').data('lobiMail');
        $mail.showCompose();
    } 
    //If mailbox view is not loaded it will be loaded as soon as the link is clicked 
    //and we attach event listener. Before mailbox is finaly initialized we change default view option 
    else {
        $(document).on('beforeInit.lobiMail.1', '.lobimail', function(ev, $mail){
            $mail.setDefaultView('compose');
            $(document).off('beforeInit.lobiMail.1', '.lobimail');
        });
    }
});
